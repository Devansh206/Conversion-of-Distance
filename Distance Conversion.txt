#include<stdio.h>
int main()
{
  float km,m,cm,ft,inch;
  printf("Enter distance between two cities");
  scanf("%f",&km);
  m=km*1000;
  cm=m*100;
  inch=cm/2.54;
  ft=inch/12;
  printf("Distance in mete=%f\n",m);
  printf("Distance in centimeter=%f",cm);
  printf("Distance in inches=%f",inch);
  printf("Distance in feet=%f",ft);
  return 0;
}
